```go
// internal/template/model.go
package template

import (
	"database/sql"
	"encoding/json"
	"time"

	"github.com/sweatshop/sweatshop/internal/shared/db"
)

// Template represents an agent template configuration
type Template struct {
	ID                string   `json:"id"`
	Name              string   `json:"name"`
	Department        string   `json:"department"`
	Description       string   `json:"description,omitempty"`
	Icon              string   `json:"icon,omitempty"`
	Prompt            string   `json:"prompt"`
	Skills            []string `json:"skills,omitempty"`
	Tools             []string `json:"tools,omitempty"`
	RuntimeType       string   `json:"runtimeType"`
	RuntimeModel      string   `json:"runtimeModel,omitempty"`
	CommunicationPeers []string `json:"communicationPeers,omitempty"`
	RequiresLead      bool     `json:"requiresLead"`
	IsBuiltIn         bool     `json:"isBuiltIn"`
	CreatedAt         time.Time `json:"createdAt"`
}

// CreateTemplateInput represents input for creating a template
type CreateTemplateInput struct {
	Name              string   `json:"name" validate:"required"`
	Department        string   `json:"department" validate:"required"`
	Description       string   `json:"description"`
	Icon              string   `json:"icon"`
	Prompt            string   `json:"prompt" validate:"required"`
	Skills            []string `json:"skills"`
	Tools             []string `json:"tools"`
	RuntimeType       string   `json:"runtimeType"`
	RuntimeModel      string   `json:"runtimeModel"`
	CommunicationPeers []string `json:"communicationPeers"`
	RequiresLead      bool     `json:"requiresLead"`
}

// UpdateTemplateInput represents input for updating a template
type UpdateTemplateInput struct {
	Name              string   `json:"name"`
	Department        string   `json:"department"`
	Description       string   `json:"description"`
	Icon              string   `json:"icon"`
	Prompt            string   `json:"prompt"`
	Skills            []string `json:"skills"`
	Tools             []string `json:"tools"`
	RuntimeType       string   `json:"runtimeType"`
	RuntimeModel      string   `json:"runtimeModel"`
	CommunicationPeers []string `json:"communicationPeers"`
	RequiresLead      *bool    `json:"requiresLead"`
}

// YAMLTemplate represents a template in YAML format
type YAMLTemplate struct {
	ID          string   `yaml:"id"`
	Name        string   `yaml:"name"`
	Department  string   `yaml:"department"`
	Description string   `yaml:"description"`
	Icon        string   `yaml:"icon"`
	Prompt      string   `yaml:"prompt"`
	Skills      []string `yaml:"skills"`
	Tools       []string `yaml:"tools"`
	Runtime     struct {
		Type  string `yaml:"type"`
		Model string `yaml:"model"`
	} `yaml:"runtime"`
	Communication struct {
		Peers       []string `yaml:"peers"`
		RequiresLead bool     `yaml:"requires_lead"`
	} `yaml:"communication"`
}

// GetAll retrieves all templates
func GetAll() ([]Template, error) {
	rows, err := db.DB.Query(`
		SELECT id, name, department, COALESCE(description, ''), COALESCE(icon, ''),
		       prompt, COALESCE(skills, '[]'), COALESCE(tools, '[]'),
		       COALESCE(runtime_type, 'claude-code'), COALESCE(runtime_model, ''),
		       COALESCE(communication_peers, '[]'), requires_lead, created_at
		FROM templates ORDER BY created_at DESC
	`)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var templates []Template
	for rows.Next() {
		var t Template
		var skillsJSON, toolsJSON, peersJSON string
		if err := rows.Scan(&t.ID, &t.Name, &t.Department, &t.Description, &t.Icon,
			&t.Prompt, &skillsJSON, &toolsJSON, &t.RuntimeType, &t.RuntimeModel,
			&peersJSON, &t.RequiresLead, &t.CreatedAt); err != nil {
			return nil, err
		}
		json.Unmarshal([]byte(skillsJSON), &t.Skills)
		json.Unmarshal([]byte(toolsJSON), &t.Tools)
		json.Unmarshal([]byte(peersJSON), &t.CommunicationPeers)
		templates = append(templates, t)
	}
	return templates, nil
}

// GetByID retrieves a template by ID
func GetByID(id string) (*Template, error) {
	var t Template
	var skillsJSON, toolsJSON, peersJSON string
	err := db.DB.QueryRow(`
		SELECT id, name, department, COALESCE(description, ''), COALESCE(icon, ''),
		       prompt, COALESCE(skills, '[]'), COALESCE(tools, '[]'),
		       COALESCE(runtime_type, 'claude-code'), COALESCE(runtime_model, ''),
		       COALESCE(communication_peers, '[]'), requires_lead, created_at
		FROM templates WHERE id = ?
	`, id).Scan(&t.ID, &t.Name, &t.Department, &t.Description, &t.Icon,
		&t.Prompt, &skillsJSON, &toolsJSON, &t.RuntimeType, &t.RuntimeModel,
		&peersJSON, &t.RequiresLead, &t.CreatedAt)
	if err == sql.ErrNoRows {
		return nil, nil
	}
	if err != nil {
		return nil, err
	}
	json.Unmarshal([]byte(skillsJSON), &t.Skills)
	json.Unmarshal([]byte(toolsJSON), &t.Tools)
	json.Unmarshal([]byte(peersJSON), &t.CommunicationPeers)
	return &t, nil
}

// Create creates a new template
func Create(t *Template) error {
	skillsJSON, _ := json.Marshal(t.Skills)
	toolsJSON, _ := json.Marshal(t.Tools)
	peersJSON, _ := json.Marshal(t.CommunicationPeers)

	_, err := db.DB.Exec(`
		INSERT INTO templates (id, name, department, description, icon, prompt, skills, tools,
		                       runtime_type, runtime_model, communication_peers, requires_lead, created_at)
		VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
	`, t.ID, t.Name, t.Department, t.Description, t.Icon, t.Prompt, skillsJSON, toolsJSON,
		t.RuntimeType, t.RuntimeModel, peersJSON, t.RequiresLead, t.CreatedAt)
	return err
}

// Update updates a template
func Update(t *Template) error {
	skillsJSON, _ := json.Marshal(t.Skills)
	toolsJSON, _ := json.Marshal(t.Tools)
	peersJSON, _ := json.Marshal(t.CommunicationPeers)

	_, err := db.DB.Exec(`
		UPDATE templates SET name = ?, department = ?, description = ?, icon = ?, prompt = ?,
		                     skills = ?, tools = ?, runtime_type = ?, runtime_model = ?,
		                     communication_peers = ?, requires_lead = ?
		WHERE id = ?
	`, t.Name, t.Department, t.Description, t.Icon, t.Prompt, skillsJSON, toolsJSON,
		t.RuntimeType, t.RuntimeModel, peersJSON, t.RequiresLead, t.ID)
	return err
}

// Delete deletes a template (only if not built-in)
func Delete(id string) error {
	_, err := db.DB.Exec("DELETE FROM templates WHERE id = ? AND is_built_in = false", id)
	return err
}
```

```go
// internal/template/loader.go
package template

import (
	"fmt"
	"io/ioutil"
	"os"
	"path/filepath"
	"time"

	"gopkg.in/yaml.v3"
	"github.com/sweatshop/sweatshop/pkg/logger"
	"github.com/sweatshop/sweatshop/pkg/uuid"
)

// Loader handles loading templates from YAML files
type Loader struct {
	templateDir string
}

// NewLoader creates a new template loader
func NewLoader(templateDir string) *Loader {
	return &Loader{templateDir: templateDir}
}

// LoadAll loads all templates from YAML files and saves to database
func (l *Loader) LoadAll() error {
	files, err := ioutil.ReadDir(l.templateDir)
	if err != nil {
		if os.IsNotExist(err) {
			logger.Info.Printf("Template directory %s does not exist, skipping", l.templateDir)
			return nil
		}
		return err
	}

	for _, file := range files {
		if filepath.Ext(file.Name()) != ".yaml" && filepath.Ext(file.Name()) != ".yml" {
			continue
		}

		path := filepath.Join(l.templateDir, file.Name())
		if err := l.loadFile(path); err != nil {
			logger.Error.Printf("Failed to load template %s: %v", path, err)
			continue
		}
	}

	return nil
}

func (l *Loader) loadFile(path string) error {
	data, err := ioutil.ReadFile(path)
	if err != nil {
		return err
	}

	var yt YAMLTemplate
	if err := yaml.Unmarshal(data, &yt); err != nil {
		return err
	}

	// Convert YAML template to Template
	t := &Template{
		ID:                 yt.ID,
		Name:               yt.Name,
		Department:         yt.Department,
		Description:        yt.Description,
		Icon:               yt.Icon,
		Prompt:             yt.Prompt,
		Skills:             yt.Skills,
		Tools:              yt.Tools,
		RuntimeType:        yt.Runtime.Type,
		RuntimeModel:       yt.Runtime.Model,
		CommunicationPeers: yt.Communication.Peers,
		RequiresLead:       yt.Communication.RequiresLead,
		IsBuiltIn:          true,
		CreatedAt:          time.Now(),
	}

	// Generate ID if not provided
	if t.ID == "" {
		t.ID = uuid.New()
	}

	// Set defaults
	if t.RuntimeType == "" {
		t.RuntimeType = "claude-code"
	}

	// Check if template already exists
	existing, err := GetByID(t.ID)
	if err != nil {
		return err
	}

	if existing != nil {
		// Update existing
		if err := Update(t); err != nil {
			return err
		}
		logger.Info.Printf("Updated template: %s", t.Name)
	} else {
		// Create new
		if err := Create(t); err != nil {
			return err
		}
		logger.Info.Printf("Loaded template: %s", t.Name)
	}

	return nil
}
```

```go
// internal/template/service.go
package template

import (
	"time"

	"github.com/sweatshop/sweatshop/pkg/uuid"
)

// Service provides template business logic
type Service struct {
	loader *Loader
}

// NewService creates a new template service
func NewService(templateDir string) *Service {
	return &Service{
		loader: NewLoader(templateDir),
	}
}

// LoadDefaults loads default templates from YAML files
func (s *Service) LoadDefaults() error {
	return s.loader.LoadAll()
}

// List returns all templates
func (s *Service) List() ([]Template, error) {
	return GetAll()
}

// Get returns a template by ID
func (s *Service) Get(id string) (*Template, error) {
	return GetByID(id)
}

// Create creates a new custom template
func (s *Service) Create(input CreateTemplateInput) (*Template, error) {
	t := &Template{
		ID:                uuid.New(),
		Name:              input.Name,
		Department:        input.Department,
		Description:       input.Description,
		Icon:              input.Icon,
		Prompt:            input.Prompt,
		Skills:            input.Skills,
		Tools:             input.Tools,
		RuntimeType:       input.RuntimeType,
		RuntimeModel:      input.RuntimeModel,
		CommunicationPeers: input.CommunicationPeers,
		RequiresLead:      input.RequiresLead,
		IsBuiltIn:         false,
		CreatedAt:         time.Now(),
	}

	if t.RuntimeType == "" {
		t.RuntimeType = "claude-code"
	}

	if err := Create(t); err != nil {
		return nil, err
	}
	return t, nil
}

// Update updates a template
func (s *Service) Update(id string, input UpdateTemplateInput) (*Template, error) {
	t, err := GetByID(id)
	if err != nil {
		return nil, err
	}
	if t == nil {
		return nil, nil
	}

	if t.IsBuiltIn {
		return nil, ErrBuiltInTemplate
	}

	if input.Name != "" {
		t.Name = input.Name
	}
	if input.Department != "" {
		t.Department = input.Department
	}
	if input.Description != "" {
		t.Description = input.Description
	}
	if input.Icon != "" {
		t.Icon = input.Icon
	}
	if input.Prompt != "" {
		t.Prompt = input.Prompt
	}
	if input.Skills != nil {
		t.Skills = input.Skills
	}
	if input.Tools != nil {
		t.Tools = input.Tools
	}
	if input.RuntimeType != "" {
		t.RuntimeType = input.RuntimeType
	}
	if input.RuntimeModel != "" {
		t.RuntimeModel = input.RuntimeModel
	}
	if input.CommunicationPeers != nil {
		t.CommunicationPeers = input.CommunicationPeers
	}
	if input.RequiresLead != nil {
		t.RequiresLead = *input.RequiresLead
	}

	if err := Update(t); err != nil {
		return nil, err
	}
	return t, nil
}

// Delete deletes a template
func (s *Service) Delete(id string) error {
	t, err := GetByID(id)
	if err != nil {
		return err
	}
	if t == nil {
		return nil
	}
	if t.IsBuiltIn {
		return ErrBuiltInTemplate
	}
	return Delete(id)
}
```

```go
// internal/template/handler.go
package template

import (
	"github.com/labstack/echo/v4"
	"github.com/sweatshop/sweatshop/internal/shared/response"
)

// Handler handles template HTTP requests
type Handler struct {
	service *Service
}

// NewHandler creates a new template handler
func NewHandler(service *Service) *Handler {
	return &Handler{service: service}
}

// List handles GET /api/templates
func (h *Handler) List(c echo.Context) error {
	templates, err := h.service.List()
	if err != nil {
		return response.InternalError(c, "Failed to list templates")
	}
	return response.OK(c, templates)
}

// Get handles GET /api/templates/:id
func (h *Handler) Get(c echo.Context) error {
	id := c.Param("id")
	template, err := h.service.Get(id)
	if err != nil {
		return response.InternalError(c, "Failed to get template")
	}
	if template == nil {
		return response.NotFound(c, "Template not found")
	}
	return response.OK(c, template)
}

// Create handles POST /api/templates
func (h *Handler) Create(c echo.Context) error {
	var input CreateTemplateInput
	if err := c.Bind(&input); err != nil {
		return response.BadRequest(c, "Invalid request body")
	}
	if input.Name == "" || input.Department == "" || input.Prompt == "" {
		return response.BadRequest(c, "Name, department, and prompt are required")
	}

	template, err := h.service.Create(input)
	if err != nil {
		return response.InternalError(c, "Failed to create template")
	}
	return response.Created(c, template)
}

// Update handles PUT /api/templates/:id
func (h *Handler) Update(c echo.Context) error {
	id := c.Param("id")
	var input UpdateTemplateInput
	if err := c.Bind(&input); err != nil {
		return response.BadRequest(c, "Invalid request body")
	}

	template, err := h.service.Update(id, input)
	if err == ErrBuiltInTemplate {
		return response.BadRequest(c, "Cannot modify built-in template")
	}
	if err != nil {
		return response.InternalError(c, "Failed to update template")
	}
	if template == nil {
		return response.NotFound(c, "Template not found")
	}
	return response.OK(c, template)
}

// Delete handles DELETE /api/templates/:id
func (h *Handler) Delete(c echo.Context) error {
	id := c.Param("id")
	err := h.service.Delete(id)
	if err == ErrBuiltInTemplate {
		return response.BadRequest(c, "Cannot delete built-in template")
	}
	if err != nil {
		return response.InternalError(c, "Failed to delete template")
	}
	return response.NoContent(c)
}

// RegisterRoutes registers template routes
func (h *Handler) RegisterRoutes(g *echo.Group) {
	g.GET("", h.List)
	g.GET("/:id", h.Get)
	g.POST("", h.Create)
	g.PUT("/:id", h.Update)
	g.DELETE("/:id", h.Delete)
}
```

```go
// internal/app/router.go
package app

import (
	"github.com/labstack/echo/v4"
	"github.com/labstack/echo/v4/middleware"
)

// SetupRouter creates and configures the Echo router
func (a *App) SetupRouter() *echo.Echo {
	e := echo.New()

	// Middleware
	e.Use(middleware.Logger())
	e.Use(middleware.Recover())
	e.Use(middleware.CORS())

	// API routes
	api := e.Group("/api")

	// Template routes (top-level)
	a.TemplateHandler.RegisterRoutes(api.Group("/templates"))

	// Team routes
	a.TeamHandler.RegisterRoutes(api.Group("/teams"))

	// Nested routes under teams
	teams := api.Group("/teams/:teamId")
	a.ProjectHandler.RegisterRoutes(teams.Group("/projects"))
	a.DepartmentHandler.RegisterRoutes(teams.Group("/departments"))
	a.TaskHandler.RegisterRoutes(teams.Group("/tasks"))

	return e
}
```

```go
// cmd/server/main.go
package main

import (
	"flag"
	"fmt"
	"os"
	"os/signal"
	"syscall"

	"github.com/sweatshop/sweatshop/internal/app"
	"github.com/sweatshop/sweatshop/internal/shared/db"
	"github.com/sweatshop/sweatshop/pkg/logger"
)

func main() {
	port := flag.Int("port", 8080, "Server port")
	dataPath := flag.String("data", "./data", "Data directory path")
	templateDir := flag.String("templates", "./configs/templates", "Templates directory path")
	flag.Parse()

	// Initialize database
	if err := db.Init(*dataPath); err != nil {
		logger.Error.Fatalf("Failed to initialize database: %v", err)
	}
	defer db.Close()
	logger.Info.Printf("Database initialized at %s", *dataPath)

	// Create application
	application := app.New(*templateDir)

	// Load default templates
	if err := application.LoadDefaultTemplates(); err != nil {
		logger.Error.Printf("Failed to load templates: %v", err)
	}

	// Setup router
	e := application.SetupRouter()

	// Start server
	go func() {
		addr := fmt.Sprintf(":%d", *port)
		logger.Info.Printf("Server starting on %s", addr)
		if err := e.Start(addr); err != nil {
			logger.Error.Printf("Server error: %v", err)
		}
	}()

	// Wait for interrupt
	quit := make(chan os.Signal, 1)
	signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
	<-quit

	logger.Info.Println("Shutting down server...")
	if err := e.Close(); err != nil {
		logger.Error.Printf("Error during shutdown: %v", err)
	}
}
```

```typescript
// web/src/features/template/api/template-api.ts
import type { Template, CreateTemplateInput, UpdateTemplateInput } from '../types/template'

const API_BASE = '/api'

export const templateApi = {
  async list(): Promise<Template[]> {
    const res = await fetch(`${API_BASE}/templates`)
    if (!res.ok) throw new Error('Failed to fetch templates')
    return res.json()
  },

  async get(id: string): Promise<Template> {
    const res = await fetch(`${API_BASE}/templates/${id}`)
    if (!res.ok) throw new Error('Failed to fetch template')
    return res.json()
  },

  async create(input: CreateTemplateInput): Promise<Template> {
    const res = await fetch(`${API_BASE}/templates`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(input),
    })
    if (!res.ok) throw new Error('Failed to create template')
    return res.json()
  },

  async update(id: string, input: UpdateTemplateInput): Promise<Template> {
    const res = await fetch(`${API_BASE}/templates/${id}`, {
      method: 'PUT',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(input),
    })
    if (!res.ok) throw new Error('Failed to update template')
    return res.json()
  },

  async delete(id: string): Promise<void> {
    const res = await fetch(`${API_BASE}/templates/${id}`, {
      method: 'DELETE',
    })
    if (!res.ok) throw new Error('Failed to delete template')
  },
}
```

**Step 3: Create template list component**

```tsx
// web/src/features/template/components/template-list.tsx
import { useEffect, useState } from 'react'
import { Plus, Trash2, Edit } from 'lucide-react'
import { Button } from '@/shared/components/ui/button'
import { Card, CardHeader, CardTitle, CardContent } from '@/shared/components/ui/card'
import { templateApi } from '../api/template-api'
import type { Template } from '../types/template'

interface TemplateListProps {
  onEdit: (template: Template) => void
  onNew: () => void
}

export function TemplateList({ onEdit, onNew }: TemplateListProps) {
  const [templates, setTemplates] = useState<Template[]>([])
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    loadTemplates()
  }, [])

  const loadTemplates = async () => {
    try {
      const data = await templateApi.list()
      setTemplates(data)
    } catch (error) {
      console.error('Failed to load templates:', error)
    } finally {
      setLoading(false)
    }
  }

  const handleDelete = async (id: string) => {
    if (!confirm('Are you sure you want to delete this template?')) return
    try {
      await templateApi.delete(id)
      setTemplates(templates.filter(t => t.id !== id))
    } catch (error) {
      console.error('Failed to delete template:', error)
    }
  }

  if (loading) {
    return <div className="p-4">Loading templates...</div>
  }

  return (
    <div className="space-y-4">
      <div className="flex items-center justify-between">
        <h2 className="text-lg font-semibold">Templates</h2>
        <Button onClick={onNew} size="sm">
          <Plus className="mr-2 h-4 w-4" />
          New Template
        </Button>
      </div>

      <div className="grid gap-4">
        {templates.map((template) => (
          <Card key={template.id}>
            <CardHeader className="flex flex-row items-center justify-between space-y-0 pb-2">
              <CardTitle className="text-base">
                <span className="mr-2">{template.icon}</span>
                {template.name}
                {template.isBuiltIn && (
                  <span className="ml-2 text-xs text-muted-foreground">(Built-in)</span>
                )}
              </CardTitle>
              <div className="flex gap-2">
                <Button
                  variant="ghost"
                  size="icon"
                  onClick={() => onEdit(template)}
                  disabled={template.isBuiltIn}
                >
                  <Edit className="h-4 w-4" />
                </Button>
                <Button
                  variant="ghost"
                  size="icon"
                  onClick={() => handleDelete(template.id)}
                  disabled={template.isBuiltIn}
                >
                  <Trash2 className="h-4 w-4" />
                </Button>
              </div>
            </CardHeader>
            <CardContent>
              <p className="text-sm text-muted-foreground">{template.description}</p>
              <div className="mt-2 flex gap-2">
                <span className="text-xs bg-secondary px-2 py-1 rounded">
                  {template.department}
                </span>
                <span className="text-xs bg-secondary px-2 py-1 rounded">
                  {template.runtimeType}
                </span>
              </div>
            </CardContent>
          </Card>
        ))}
      </div>
    </div>
  )
}
```
